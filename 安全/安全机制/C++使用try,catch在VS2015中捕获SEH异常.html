#include <iostream>

int main()
{
	try
	{
		char* pch;
		pch = (char*)00001234;   //给予一个非法地址 
		*pch = 6; //对非法地址赋值，会造成Access Violation 异常 
		std::cout << "It is OK !" << std::endl;
	}
	catch(...)
	{
		std::cerr << "catched" << std::endl;
	}

	getchar();

    return 0;
}
